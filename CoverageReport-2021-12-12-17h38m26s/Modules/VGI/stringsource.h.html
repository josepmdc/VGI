class StringCharSource {

  StringCharSource(const char* str, std::size_t size)
      : m_str(str), m_size(size), m_offset(0) {}

    StringCharSource source(*this);

      source.m_offset += i;
    else

    return source;
  }

    ++m_offset;
    return *this;
  }

    return *this;
